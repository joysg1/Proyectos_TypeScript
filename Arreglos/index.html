Arreglos en TypeScript

Código TypeScript Ejemplo

// Declaración de un arreglo en TypeScript
let miArreglo: number[] = [10, 20, 30];

// Agregar elemento al final
miArreglo.push(40);

// Eliminar último elemento
miArreglo.pop();

// Obtener longitud del arreglo
let longitud = miArreglo.length; // 3

// Acceder a un elemento por índice
let primerElemento = miArreglo[0]; // 10
